Вот минимальный пример Node.js‑бэкенда для инвентаря, который можно быстро поднять и подключить к Unity через HTTP‑запросы. Он использует Express и хранит данные в памяти (для продакшна замените на базу — MongoDB, SQLite и т.п.).

```js
// server.js
const express = require('express');
const cors = require('cors');
const app = express();
const PORT = 3000;

app.use(cors());               // разрешаем запросы с любого домена (для Unity)
app.use(express.json());        // парсим JSON‑тело запросов

// Внутреннее хранилище – массив предметов
let inventory = [];
// Пример структуры предмета { id: 1, name: "Wood", count: 5 }
let nextId = 1;

// Получить весь инвентарь
app.get('/api/inventory', (req, res) => {
  res.json(inventory);
});

// Добавить новый предмет (или увеличить количество, если уже есть)
app.post('/api/inventory', (req, res) => {
  const { name, count } = req.body;
  if (!name || typeof count !== 'number') {
    return res.status(400).json({ error: 'Invalid payload' });
  }
  // Ищем, есть ли уже такой предмет
  const existing = inventory.find(it => it.name === name);
  if (existing) {
    existing.count += count;
    return res.json(existing);
  }
  const newItem = { id: nextId++, name, count };
  inventory.push(newItem);
  res.status(201).json(newItem);
});

// Изменить количество конкретного предмета
app.put('/api/inventory/:id', (req, res) => {
  const id = parseInt(req.params.id);
  const { count } = req.body;
  const item = inventory.find(it => it.id === id);
  if (!item) return res.status(404).json({ error: 'Item not found' });
  if (typeof count !== 'number') return res.status(400).json({ error: 'Invalid count' });
  item.count = count;
  res.json(item);
});

// Удалить предмет
app.delete('/api/inventory/:id', (req, res) => {
  const id = parseInt(req.params.id);
  const idx = inventory.findIndex(it => it.id === id);
  if (idx === -1) return res.status(404).json({ error: 'Item not found' });
  const removed = inventory.splice(idx, 1)[0];
  res.json(removed);
});

app.listen(PORT, () => {
  console.log(`Inventory server listening on http://localhost:${PORT}`);
});
```

1. **Тестируй локально** – запусти `node server.js` и проверяй через `curl` или Postman.
